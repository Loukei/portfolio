132 QNetworkReply* GDrive::GDriveFileResumableUpdate::requestInitialSession(const QUrl &url)
133 {
134  qInfo() << Q_FUNC_INFO;
135  // collect file info
136  const QString fileName = QFileInfo(m_file.fileName()).baseName();
137  const QString fileMimeType = QMimeDatabase().mimeTypeForFile(m_file).name();
138  const qint64 fileSize = m_file.size();
139  // Build request
140  QNetworkRequest request(url);
141  const QByteArray body = QString("{\"name\":\"%1\"}").arg(fileName).toUtf8();
142  request.setHeader(QNetworkRequest::ContentLengthHeader,body.size());
143  request.setHeader(QNetworkRequest::ContentTypeHeader,"application/json; charset=UTF-8");
144  request.setRawHeader("Authorization",
145  QString("Bearer %1").arg(mp_google->token()).toLatin1());
146  request.setRawHeader("X-Upload-Content-Type",fileMimeType.toLatin1());
147  request.setRawHeader("X-Upload-Content-Length",QByteArray::number(fileSize));
148  // Send request
149  auto reply = mp_google->networkAccessManager()->sendCustomRequest(request,"PATCH",body);
150  connect(reply,&QNetworkReply::finished,
151  this,&GDriveFileResumableUpdate::onInitialSession_ReplyFinished);
152  connect(reply,QOverload<QNetworkReply::NetworkError>::of(&QNetworkReply::error),
153  this,&GDriveFileResumableUpdate::onInitialSession_ReplyError);
154  /* emit uploadProgress signal,so ui like progressDialog will be open */
155  emit uploadProgress(0,m_file.size());
